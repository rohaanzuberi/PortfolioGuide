WITH past_employees AS (
    SELECT 
        emp_no,
        Min(from_date) AS start_date,
        Max(to_date) AS end_date
    FROM salaries
    WHERE to_date != (SELECT MAX(to_date) FROM salaries)
    GROUP BY emp_no),
    
employee_tenures AS (
    SELECT 
        emp_no,
        start_date,
        end_date,
        DATEDIFF(end_date, start_date) / 365 AS employee_tenure
    FROM past_employees)
​
SELECT 
    title,
    AVG(employee_tenure) AS avg_employee_tenure
FROM employee_tenures
INNER JOIN titles
    ON employee_tenures.emp_no = titles.emp_no
    AND employee_tenures.end_date = titles.to_date
GROUP BY title
ORDER BY avg_employee_tenure

How many employees were hired between 1985 and 2001 and what was the gender diversity of recruitment?

SELECT
    YEAR(de.from_date) AS Recruitment_Year,
    COUNT(e.emp_no) AS Employees_Hired,
    e.gender
FROM
    dept_emp AS de
INNER JOIN
    employees as e
ON
    de.emp_no = e.emp_no
WHERE
    YEAR(de.from_date) >= "1985"
AND
    YEAR(de.from_date) <= "2001"
GROUP BY
    Recruitment_Year, e.gender
ORDER BY
    Recruitment_Year;
